# Array.prototype.map

```js
function _map(arr, cb) {
    // validate
    if(!Array.isArray(arr) || !arr.length || typeof cb !== 'function'){
        return []
    }

    let result = []     //创建新数组进行操作不影响原始数组
    for(let i = 0, len = arr.length; i < len; i++ ){
        result.push(cb(arr[i], i, arr))
    }
    return result
}
```

# Array.prototype.filter

```js
function _filter(arr, cb) {
    //validate
    if(!Array.isArray(arr) || !arr.length || typeof cb !== 'function'){
        return []
    }
    let result = []
    for(let i = 0, len = arr.length; i < len; i++ ){
      if (cb(arr[i], i, arr)) { 
      // 如果条件为真，则将数组元素 push 到 result 中
        result.push(arr[i]);
      }
    }
    return result; 
}
```

# Array.prototype.reduce

```js
function _reduce(arr, cb, initialValue) {
  // validate
  if (!Array.isArray(arr) || !arr.length || typeof cb !== 'function'){
    return [];
  } 
    // 如果没有将initialValue传递给该函数，我们将使用第一个数组项作为initialValue
    let value = initialValue ? initialValue : arr[0];
    
    // 如果有传递 initialValue，则索引从 1 开始，否则从 0 开始
    for (let i = initialValue ? 0 : 1, len = arr.length; i < len; i++) {
        value = cb(value, arr[i], i, arr); 
    }
    return value;
  
}
```

# 实现一个函数缓存
```js
const slice = Array.prototype.slice;
function memoize(fn) {
  const cache = {};
  return (...args) => {
    const params = slice.call(args);
    console.log(params);
    if (cache[params]) {
      console.log('cached');
      return cache[params];
    } else {
      let result = fn(...args);
      cache[params] = result;
      console.log(`not cached`);
      return result;
    }
  }
}
const makeFullName = (fName, lName) => `${fName} ${lName}`;
const reduceAdd = (numbers, startingValue = 0) => numbers.reduce((total, cur) => total + cur, startingValue);

const memoizedMakeFullName = memoize(makeFullName);
const memoizedReduceAdd = memoize(reduceAdd);

memoizedMakeFullName("Marko", "Polo");
memoizedMakeFullName("Marko", "Polo");

memoizedReduceAdd([1, 2, 3, 4, 5], 5);
memoizedReduceAdd([1, 2, 3, 4, 5], 5);

```
